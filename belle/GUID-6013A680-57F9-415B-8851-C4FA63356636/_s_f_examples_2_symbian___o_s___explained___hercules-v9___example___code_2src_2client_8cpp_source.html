00022 // Runs client-side and starts the separate server process
00023 static TInt StartTheServer()
00024         {
00025         RProcess server;
00026         TInt r=server.Create(KServerBinaryName, KNullDesC);
00027         if (r!=KErrNone)
00028                 return r;
00029         
00030         TRequestStatus stat;
00031         server.Rendezvous(stat);
00032         if (stat!=KRequestPending)
00033                 server.Kill(0);         // abort startup
00034         else
00035                 server.Resume();        // logon OK - start the server
00036         
00037         User::WaitForRequest(stat);             // wait for start or death
00038         
00039         // Check the exit type.
00040         // We can't use the 'exit reason' because if the server panicked this
00041         // is set to the panic 'reason' (which may be '0' and cannot thus be distinguished
00042         // from KErrNone)
00043         r = server.ExitType();
00044         if (EExitPanic==r)
00045                 r = KErrGeneral; 
00046         else
00047                 r = stat.Int();
00048         
00049         server.Close(); // This is no longer needed
00050         return r;
00051         }

00055 EXPORT_C TInt RHerculesSession::Connect()
00056         {
00057         TInt retry=2;
00058         for (;;)
00059                 {// Uses system-pool message slots
00060                 TInt r=CreateSession(KServerName,TVersion(1,0,0));
00061                 if ( (KErrNotFound!=r) && (KErrServerTerminated!=r) )
00062                         return (r);
00063                 if (--retry==0)
00064                         return (r);
00065                 r=StartTheServer();
00066                 if ( (KErrNone!=r) && (KErrAlreadyExists!=r) )
00067                         return (r);
00068                 }
00069         }
