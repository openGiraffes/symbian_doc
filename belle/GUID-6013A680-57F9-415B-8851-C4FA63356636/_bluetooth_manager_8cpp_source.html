00202 Send the RPS's player choice to the remote BT device. On completion calls SendDataComplete()
00203 ============================================================================

00205 void CBluetoothManager::SendData(TRoshambo::TElement aElement)
00206         {
00207         //Note that the code here to send the data to the remote device is the same for the 
00208         //Master and slave. Change the code to be like ReplayGame() above if your multiplayer game needs to handle

00213                 rpsData().iElement = aElement;

00215 
00216                 //Since the sending is an asynchronous operation we need to cache the local player's choice
00217                 iGameScreenMgr.Roshambo().SetLocalPlayer(aElement);

00220                 iBtConnBase->SendData(rpsData);

00240         {
00241         iGameScreenMgr.SetGameState(CGameScreenManager::EConnectingScreen);
00242         iBtDevAddr = aResponse().BDAddr();
00243         
00244         if(iBtConnBase)
00245                 {
00246                 iBtConnBase->Cancel();

00251         //It's important to note that in a multiplayer game with more then two players, the engine needs to create an

00253         //the BT devices (Slaves) selected by the player (Master). RHostResolver needs to be use instead of RNotifier

00258         TRAPD(err, StartConnectorL());

00260                 {
