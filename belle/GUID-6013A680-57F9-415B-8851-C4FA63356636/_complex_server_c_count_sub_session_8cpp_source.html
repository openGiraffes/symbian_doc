00055 void CCountSubSession::SetFromStringL(const RMessage2& aMessage)
00056         {
00057            // length of passed descriptor (1st parameter passed from client)
00058         TInt deslen = aMessage.GetDesLength(0);
00059         
00060           // Passed data will be saved in this descriptor.
00061     RBuf buffer;
00062       
00063       // Max length set to the value of "deslen", but current length is zero
00064     buffer.CreateL(deslen);
00065       
00066       // Do the right cleanup if anything subsequently goes wrong
00067     buffer.CleanupClosePushL();
00068     
00069       // Copy the client's descriptor data into our buffer.
00070     aMessage.ReadL(0,buffer,0);
00071     
00072       // Now do a validation to make sure that the string only has digits
00073     if (buffer.Length() == 0)
00074         {
00075         User::Leave(ENonNumericString);
00076         }
00077     
00078     TLex16 lexer;
00079     
00080     lexer.Assign(buffer);
00081     while (!lexer.Eos())
00082         {
00083         TChar thechar;
00084         
00085         thechar = lexer.Peek();
00086         if (!thechar.IsDigit())
00087             {
00088                 User::Leave(ENonNumericString);
00089             }
00090         lexer.Inc();
00091         }
00092        
00093       // Convert to a simple TInt value. 
00094     lexer.Assign(buffer);           
00095     if (lexer.Val(iCount))
00096         {
00097         User::Leave(ENonNumericString);
00098         }
00099     
00100           // Clean up the memory acquired by the RBuf variable "buffer"
00101         CleanupStack::PopAndDestroy();
00102         }
