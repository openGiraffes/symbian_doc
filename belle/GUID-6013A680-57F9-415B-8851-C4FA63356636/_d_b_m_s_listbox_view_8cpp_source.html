00163 // Specify locations and sizes of the label and listbox. Called by

00165 // ---------------------------------------------------------------------------
00166 //
00167 void CDBMSListboxView::SizeChanged()
00168     {
00169 
00170         TRect containerRect = Rect(); // Rect of the container
00171 
00172         TSize labelSize = iLabel->MinimumSize();
00173 
00174         // Show the label in the upper part of the screen and in whole
00175         // screen width
00176         iLabel->SetExtent( TPoint(5,2),
00177                            TSize(containerRect.Width(), labelSize.iHeight) );
00178 
00179         // Fill the rest (bottom) of the screen with listbox
00180         iListBox->SetExtent( TPoint(5,labelSize.iHeight + 5 ),
00181             TSize( containerRect.Width(),
00182                    containerRect.Height()-labelSize.iHeight ) );
00183                    
00184     }
