00128 LOCAL_C void callExampleL() 
00129     {
00130           // Initialize and call the example code under cleanup stack.
00131         _LIT(KMsgExampleCode,"Symbian platform Example Code");

00158 //
00159 // Do the example
00160 //
00161 // Example includes a NewL() function of the CExample class which may leave
00162 //
00164 void doExampleL()
00165         {
00166           // Memory alloc fails on the 'allocFailNumber' attempt. 
00167         __UHEAP_SETFAIL(RHeap::EDeterministic,allocFailNumber);
00168           // Allocate - leave if allocation fails.
00169         CExample* myExample = CExample::NewL(5);
00170           // Push onto the cleanup stack, if successful
00171         CleanupStack::PushL(myExample);
00172           // Do something that can leave.
00173         myExample->DoSomethingL();
00174           // Pop from the cleanup stack and destroy
00175         CleanupStack::PopAndDestroy();
00176         }
