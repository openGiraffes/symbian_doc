00041 * MsgQReceive (qName, msg, maxNBytes, timeout, err)
00042 * Description: Function for receiving a message with internal copy
00043 ********************************************************************************/

00045 EXPORT_C int MsgQReceive(ULONG qName, char* msg, ULONG maxNBytes, int timeout, int* err) {
00046         MSGQ_INFO* pMsgQInfo = NULL;
00047         ssize_t rxBytes;
00048         int toTicks;
00049 
00050         struct {
00051                 long  mtype;
00052                 char mtext[MAX_MSG_LEN];
00053         } message;
00054 
00055         /* check parameters */
00056         if ((pMsgQInfo = MsgQTableLookup(qName)) != NULL) {
00057                 if(timeout != NO_WAIT && timeout != WAIT_FOREVER)
00058                         toTicks = timeout < 10 ? 1: (ULONG)((timeout * sysconf(_SC_CLK_TCK)) / 1000);
00059                 else
00060                         toTicks = timeout;      
00061 
00062                 message.mtype = 1;
00063                 /* receive message */
00064                 if((rxBytes = msgrcv(pMsgQInfo->qId, &message, (size_t)maxNBytes, 1, toTicks)) != -1) {
00065                         message.mtext[rxBytes-4] = '\0';
00066                         bcopy(message.mtext, msg, rxBytes);
00067                         pMsgQInfo->numMsgs--;
00068                         *err = OK;
00069                         return ((ULONG)rxBytes);
00070                 }
00071                 else
00072                         *err = errno;
00073         }
00074         else
00075                 *err = KMsgQLibQIdErr;
00076 
00077         return(ERROR);
00078 }
