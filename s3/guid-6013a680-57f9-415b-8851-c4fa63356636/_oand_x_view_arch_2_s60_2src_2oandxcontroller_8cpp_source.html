00138                 iNumCrossWins += 1;

00140         default:
00141                 break;

00143         for (TInt i=KNumHistoryRecords-1; i>0; i--)
00144                 {
00145                 iGameRecords[i] = iGameRecords[i-1];
00146                 }
00147         iGameRecords[0]=iLastGameResult;
00148         iLastGameResult = ETileDraw;
